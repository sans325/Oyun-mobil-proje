const GameConfig = {
  WIDTH: 400,
  HEIGHT: 600,
  GRAVITY: 0.4,
  MAX_DROP_SPEED: 12,
  JUMP_POWER: -9,
  PIPE_WIDTH: 60,
  GAP_SIZE: 120,
  PIPE_SPEED: 5
};

class Bird {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.velocity = 0;
    this.wingAnimation = 0;
  }

  update() {
    this.velocity += 
      GameConfig.GRAVITY;
    this.y += this.velocity;
  }

  jump() {
    this.velocity = 
      GameConfig.JUMP_POWER;
  }

  draw(ctx) {
    // Kuş çizimi
    ctx.fillStyle = '#FFFF00';
    ctx.beginPath();
    ctx.arc(this.x + 20, 
            this.y + 20, 
            15, 0, 
            Math.PI * 2);
    ctx.fill();
  }
}

class Pipe {
  constructor(x, height) {
    this.x = x;
    this.pipeHeight = height;
  }

  checkCollision(bird) {
    return bird.x + 
      24 > this.x;
  }
}

// Oyun başlat

  }
