// Flappy Bird - Web Oyunu
const GameConfig = {
  WIDTH: 400,
  HEIGHT: 600,
  GRAVITY: 0.4,
  MAX_DROP_SPEED: 12,
  JUMP_POWER: -9,
  PIPE_WIDTH: 60,
  GAP_SIZE: 120
};

class Bird {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.velocity = 0;
  }

  jump() {
    this.velocity = 
      GameConfig.JUMP_POWER;
  }

  update() {
    this.velocity += 
      GameConfig.GRAVITY;
    this.y += this.velocity;
  }
}

class Pipe {
  constructor(x, height) {
    this.x = x;
    this.pipeHeight = height;
    this.width = 60;
  }

  checkCollision(bird) {
    return bird.x + 
      24 > this.x &&
      (bird.y < this.pipeHeight ||
       bird.y > this.pipeHeight + 
       120);
  }
}
